const asyncIterator = {
  [Symbol.asyncIterator]() {
    return {
      i: 0,
      next() {
        if (this.i < 5)
          return Promise.resolve({ value: this.i++, done: false });

        return Promise.resolve({ done: true });
      },
    };
  },
};

(async function() {
  for await (let value of asyncIterator) {
    console.log(value);
  }
})();
