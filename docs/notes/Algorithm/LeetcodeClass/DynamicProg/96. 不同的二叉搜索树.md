# 96. 不同的二叉搜索树

```js
/*
 * @lc app=leetcode.cn id=96 lang=javascript
 *
 * [96] 不同的二叉搜索树
 */

// @lc code=start
/**
 * @param {number} n
 * @return {number}
 */
var numTrees = function (n) {
  const dp = Array(n + 1).fill(0);
  // 边界条件
  dp[0] = 1;
  dp[1] = 1;
  // 我们将大问题分解成小问题
  // 外层循环目的是为了求出最后n的不同二叉树总数 n的结果依赖于n-1, 以此类推 n-2 n-3 ... 2, 1, 0
  // 那每个n都是可以是总长度, 在这中间某个值是可以为根的 所需要循环判断每个可能作为根的结果
  // 从而推断出 dp[i] 的值依赖于根节点j, 也就是
  // dp[i] = dp[j-1] * dp[i-j] 也就是根节点左右的长度下的总数相乘
  for (let i = 2; i <= n; i++) {
    for (let j = 1; j <= i; j++) {
      dp[i] += dp[j - 1] * dp[i - j];
    }
  }

  return dp[n];
};
// @lc code=end

console.log(numTrees(3)); // 5
console.log(numTrees(2)); // 2
console.log(numTrees(1)); // 1
```
