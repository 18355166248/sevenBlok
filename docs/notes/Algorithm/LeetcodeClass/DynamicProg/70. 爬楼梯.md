  var climbStairs = function (n) {
  const map = new Map();

return recursion(n);

function recursion(n) {
if (n <= 2) return n;

    let n1 = map.get(n - 1);
    let n2 = map.get(n - 2);
    if (!n1) {
      n1 = recursion(n - 1);
      map.set(n - 1, n1);
    }
    if (!n2) {
      n2 = recursion(n - 2);
      map.set(n - 2, n2);
    }
    return n1 + n2;

}
};
// @lc code=end

console.log(climbStairs(4));
