# 300. 最长递增子序列

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var lengthOfLIS = function (nums) {
  // 定义一个dp数组, 每个索引跟nums数组一一对应
  // 默认第一个最长子序列是1 因为前面没有值了
  const dp = [1];
  // 缓存最大值 每次外循环完之后取最大值
  let max = dp[0];
  for (let i = 1; i < nums.length; i++) {
    for (let j = 0; j < i; j++) {
      // 内循环 从开始往后一直遍历到i-1
      if (nums[i] > nums[j]) {
        // 判断当前值是否大于上一个值
        if (dp[i]) {
          // 如果当前值存在且大于前面的值且大与当前值 更新当前值
          if (dp[j] + 1 > dp[i]) {
            dp[i] = dp[j] + 1;
          }
        } else {
          // 当前值不存在且大于前面的值 直接赋值为前面的值+1
          dp[i] = dp[j] + 1;
        }
      } else if (!dp[i]) {
        // 当前值比前面的值大且不存在 直接设置为1
        dp[i] = 1;
      }
    }
    max = Math.max(dp[i], max);
  }
  // 取最大值
  return max;
};
```
