# 64. 最小路径和

```js
/*
 * @lc app=leetcode.cn id=64 lang=javascript
 *
 * [64] 最小路径和
 */

// @lc code=start
/**
 * @param {number[][]} grid
 * @return {number}
 */
var minPathSum = function (grid) {
  // 生成dp
  const dp = Array.from({ length: grid.length }, () =>
    Array(grid[0].length).fill(0)
  );
  // 两层循环
  for (let i = 0; i < grid.length; i++) {
    for (let j = 0; j < grid[0].length; j++) {
      if (i === 0) {
        // 边界情况 遍历到第一行 不会存在上面有值 只用考虑左侧即可
        dp[i][j] = (dp[i][j - 1] || 0) + grid[i][j];
      } else if (j === 0) {
        // 边界情况 遍历到第一列 不会存在左侧有值 只用考虑上面即可
        dp[i][j] = (dp[i - 1] ? dp[i - 1][j] : 0) + grid[i][j];
      } else {
        // 遍历到非最左最上的话 直接比较上面和左右的最小值 加上当前值就是最小路径和
        dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - 1]) + grid[i][j];
      }
    }
  }
  return dp[grid.length - 1][grid[0].length - 1];
};
// @lc code=end

console.log(
  minPathSum([
    [1, 3, 1],
    [1, 5, 1],
    [4, 2, 1],
  ])
); // 7

console.log(
  minPathSum([
    [1, 2, 3],
    [4, 5, 6],
  ])
); // 12
```
