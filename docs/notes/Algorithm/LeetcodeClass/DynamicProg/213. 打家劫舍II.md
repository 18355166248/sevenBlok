# 213. 打家劫舍 II

```javascript
// @lc code=start
/**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function (nums) {
  // 首尾不能同时 所以找到 0, n-1 和  1-n 的最大值 取最大值就是结果
  const len = nums.length;
  if (len === 1) return nums[0];
  if (len === 2) return nums[0] > nums[1] ? nums[0] : nums[1];

  const one = getNums(nums, 0, len - 1);
  const two = getNums(nums, 1, len);
  return Math.max(one, two);
};

function getNums(nums, s, e) {
  const len = e - s;
  const dp = Array(len).fill(0);
  dp[0] = nums[s];
  dp[1] = nums[s] > nums[s + 1] ? nums[s] : nums[s + 1];
  for (let i = 2; i < len; i++) {
    // 这里要注意取到的nums 的索引要基于开始值s累加
    dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i + s]);
  }
  return dp[len - 1];
}
// @lc code=end

console.log(rob([2, 3, 2])); // 3
console.log(rob([1, 2, 3, 1])); // 4
console.log(rob([1, 2, 3])); // 3
```
