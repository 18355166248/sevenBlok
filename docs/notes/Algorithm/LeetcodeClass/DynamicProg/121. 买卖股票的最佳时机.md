# 121. 买卖股票的最佳时机

```js
/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  // 找到整个列表的最小值
  let min = Infinity;
  // 找到整个列表的最大值 计算差值
  let max = 0;
  prices.forEach((price) => {
    if (price < min) min = price;
    if (price - min > max) {
      max = price - min;
    }
  });
  return max;
};
```
