# 44. 通配符匹配

```javascript
/*
 * @lc app=leetcode.cn id=44 lang=javascript
 *
 * [44] 通配符匹配
 */

// @lc code=start
/**
 * @param {string} s
 * @param {string} p
 * @return {boolean}
 */
var isMatch = function (s, p) {
  const m = s.length;
  const n = p.length;
  const dp = Array.from({ length: m + 1 }, () => Array(n + 1).fill(false));
  dp[0][0] = true;
  // 判断第一个星号对应的0字符 dp为true 后面的星就不准了 所以停止循环
  for (let j = 1; j <= n; j++) {
    if (p.charAt(j - 1) == "*") {
      dp[0][j] = true;
    } else {
      break;
    }
  }
  // 内外循环从1开始 表示从字符的起点开始
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (p[j - 1] === "*") {
        // 如果为星号 则表示要么匹配0次或者多次
        // 匹配0次的话那就取上个字符模式[i][j-1]
        // 匹配多个的话那就取上个字符[i-1][j]
        dp[i][j] = dp[i][j - 1] || dp[i - 1][j];
      } else if (p.charAt(j - 1) === "?" || s[i - 1] === p[j - 1]) {
        // 如果不是星号那就是问号或者小写字母 如果是问号那直接匹配 或者如果字符和字符模式的对应索引匹配 那取上一个字符和上一个的字符模式的结果即可
        dp[i][j] = dp[i - 1][j - 1];
      }
    }
  }
  // console.log(dp);
  return dp[m][n];
};

// @lc code=end

console.log(isMatch("aa", "a")); // false
console.log(isMatch("aa", "*")); // true
console.log(isMatch("cb", "?a")); // false
console.log(isMatch("adceb", "*a*b")); // true
console.log(isMatch("aab", "c*a*b")); // false
```

- 时间复杂度 O(m\*n)
- 空间复杂度 O(m\*n)
