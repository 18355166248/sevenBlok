# 279. 完全平方数

```javascript
/*
 * @lc app=leetcode.cn id=279 lang=javascript
 *
 * [279] 完全平方数
 */

// @lc code=start
/**
 * @param {number} n
 * @return {number}
 */
var numSquares = function (n) {
  const dp = Array(n + 1).fill(0);
  // 外循环循环数字
  for (let i = 1; i <= n; i++) {
    let min = Number.MAX_VALUE;
    // 内循环起始为数字1 截止为 j的平方小于等于i
    // 也就是对应的dp[j]的值依赖于 i-(j的平方) 对应的最小值
    for (let j = 1; j * j <= i; j++) {
      min = Math.min(min, dp[i - j * j]);
    }
    // 拿着最小值+1就是当前数字所需要的最小和
    dp[i] = min + 1;
  }
  // console.log(dp);
  return dp[n];
};
// @lc code=end
```
