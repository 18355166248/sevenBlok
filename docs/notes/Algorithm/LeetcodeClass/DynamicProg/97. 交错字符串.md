# 97. 交错字符串

```js
/*
 * @lc app=leetcode.cn id=97 lang=javascript
 *
 * [97] 交错字符串
 */

// @lc code=start
/**
 * @param {string} s1
 * @param {string} s2
 * @param {string} s3
 * @return {boolean}
 */
var isInterleave = function (s1, s2, s3) {
  const m = s1.length;
  const n = s2.length;
  // 长度不一致 表示无法交错合并 直接返回false
  if (m + n !== s3.length) return false;
  // 表示前i个字符和前j个字符交错合并是否满足s3[i+j]的字符
  const dp = Array.from({ length: m + 1 }, () => Array(n + 1).fill(false));
  // 0字符串用合并能合成空字符串
  dp[0][0] = true;

  for (let i = 0; i <= m; i++) {
    for (let j = 0; j <= n; j++) {
      // 一种是i>0 一种是j>0
      // 大问题细分为小问题

      if (i > 0) {
        // dp[1][0] 取决于 dp[0][0] 和 s1[0] === s3[0]
        // dp[1][1] 取决于 dp[0][1] 和 s1[0] === s3[1]
        // dp[1][2] 取决于 dp[0][2] 和 s1[0] === s3[2]
        // ...
        // dp[m][n] 取决于 dp[i-1][j] 和 s1[i-1] === s3[i+j-1]
        dp[i][j] = dp[i][j] || (dp[i - 1][j] && s1[i - 1] === s3[i - 1 + j]);
      }
      if (j > 0) {
        // dp[0][1] 取决于 dp[0][0] 和 s2[0] === s3[0]
        // dp[1][1] 取决于 dp[1][0] 和 s2[0] === s3[1]
        // dp[2][1] 取决于 dp[2][0] 和 s2[0] === s3[2]
        // ...
        // dp[m][n] 取决于 dp[i][j-1] 和 s2[j-1] === s3[i+j-1]
        dp[i][j] = dp[i][j] || (dp[i][j - 1] && s2[j - 1] === s3[j - 1 + i]);
      }
    }
  }
  // console.log(dp);
  return dp[m][n];
};
// @lc code=end

console.log(isInterleave("aabcc", "dbbca", "aadbbcbcac")); // true

console.log(isInterleave("", "", "")); // true
```
