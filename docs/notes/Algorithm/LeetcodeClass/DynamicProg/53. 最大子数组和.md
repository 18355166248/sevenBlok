# 53.最大子数组和

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function (nums) {
  if (nums.length === 1) return nums[0];

  let max = nums[0];
  // 缓存前一个的最大值
  let pre = 0;
  nums.forEach((num) => {
    // 重点在这里
    // 计算当前的最大值, 当前索引的最大值有两种情况, 一种就是当前的数比前一个的最大值还大
    // 另外一种就当前数加上前一个的最大值 拿到当前索引的最大值赋值个pre
    // 比较历史的最大值和当前索引的最大值, 返回较大的
    pre = Math.max(pre + num, num);
    max = Math.max(pre, max);
  });
  return max;
};
```
