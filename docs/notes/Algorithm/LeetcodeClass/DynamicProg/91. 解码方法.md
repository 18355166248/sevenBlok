# 91. 解码方法

```javascript
/*
 * @lc app=leetcode.cn id=91 lang=javascript
 *
 * [91] 解码方法
 */

// @lc code=start
/**
 * @param {string} s
 * @return {number}
 */
var numDecodings = function (s) {
  const len = s.length;
  // 初始化dp 长度为字符串长度+1
  const dp = Array(len + 1).fill(0);
  // 空字符串可以有 111 种解码方法，解码出一个空字符串
  dp[0] = 1;
  // 循环字符串长度 从1开始到len
  for (let i = 1; i <= len; i++) {
    // 两种情况 一种是前一个字符串匹配一次 那么等同前一个即可
    if (s[i - 1] !== "0") {
      dp[i] += dp[i - 1];
    }

    // 第二种 我们使用了2个字符 即使用了 s[i-2]和s[i-1]进行编码 前提条件是 i>1 s[i-2]不能等于0 两个数的总和要小于等于26
    if (i > 1 && s[i - 2] !== "0" && s[i - 2] * 10 + Number(s[i - 1]) <= 26) {
      dp[i] += dp[i - 2];
    }
  }
  return dp[len];
};
// @lc code=end

console.log(numDecodings("12")); // 2
console.log(numDecodings("226")); // 3
console.log(numDecodings("06")); // 0
console.log(numDecodings("0")); // 0
```
