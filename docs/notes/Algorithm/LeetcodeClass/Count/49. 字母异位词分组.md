# 49. 字母异位词分组

```js
/*
 * @lc app=leetcode.cn id=49 lang=javascript
 *
 * [49] 字母异位词分组
 */

// @lc code=start
/**
 * @param {string[]} strs
 * @return {string[][]}
 */
var groupAnagrams = function (strs) {
  // 缓存map 记录结果
  const map = new Map();

  // 遍历外层数组
  for (let i = 0; i < strs.length; i++) {
    // 初始化26个字母长度的数组 用于后续更新每个字符出现的次数 默认0
    const letters = Array(26).fill(0);
    // 遍历每个字符
    for (let j = 0; j < strs[i].length; j++) {
      // 默认拿到的a的字符数字是97 对应每个字母减去97就能跟letters的索引对上
      const code = strs[i][j].charCodeAt() - 97;
      // 更新出现次数
      letters[code] += 1;
    }
    // 将letters出现次数转成字符串 并通过字符串判断当前字符是否存在或者没有存在从而将原字符更新进map中存储下来
    const letterStr = letters.toString();
    if (map.has(letterStr)) {
      map.get(letterStr).push(strs[i]);
    } else {
      map.set(letterStr, [strs[i]]);
    }
  }
  // map转数组即为结果
  return [...map.values()];
};
// @lc code=end

console.log(groupAnagrams(["eat", "tea", "tan", "ate", "nat", "bat"])); // [["bat"],["nat","tan"],["ate","eat","tea"]]

console.log(groupAnagrams([""])); // [[""]]
```
