# 33. 搜索旋转排序数组

```js
/*
 * @lc app=leetcode.cn id=33 lang=javascript
 *
 * [33] 搜索旋转排序数组
 */

// @lc code=start
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function (nums, target) {
  const len = nums.length;
  if (len === 1) return nums[0] === target ? 0 : -1;
  let left = 0;
  let right = len - 1;
  while (left <= right) {
    const mid = (left + right) >> 1;
    if (nums[mid] === target) return mid;
    if (nums[left] <= nums[mid]) {
      // 左侧存在升序
      if (nums[left] <= target && target < nums[mid]) {
        // 第一个值小于等于目标值且目标值小于中间值 则目标所在的区间在 [l, mid-1]
        right = mid - 1;
      } else {
        // 反之目前所在区间为 [mid+1, r]
        left = mid + 1;
      }
    } else {
      // 右侧存在升序
      if (nums[mid] < target && target <= nums[len - 1]) {
        // 中间值小于目标值且目标值小于等于最后一个值 则目标所在的区间在 [mid+1, r]
        left = mid + 1;
      } else {
        // 反之目前所在区间为 [l, mid-1]
        right = mid - 1;
      }
    }
  }
  return -1;
};
// @lc code=end

console.log(search([4, 5, 6, 7, 0, 1, 2], 0)); // 4
console.log(search([5, 1, 3], 0)); // 4
```
